110void RayboyUI::renderTile(const uint8_t *tile_data, int tile_num, int x,
111 int y) {
112 uint16_t tile_offset = tile_num * 16;
113
114 if (tile_offset + 15 >= 0x1800) {
115 return;
116 }
117
118 for (int tile_y = 0; tile_y < 8; tile_y++) {
119 uint16_t byte_offset = tile_offset + (tile_y * 2);
120 uint8_t byte1 = tile_data[byte_offset];
121 uint8_t byte2 = tile_data[byte_offset + 1];
122
123 for (int tile_x = 0; tile_x < 8; tile_x++) {
124 int bit = 7 - tile_x;
125 uint8_t color_id = ((byte2 >> bit) & 1) << 1 | ((byte1 >> bit) & 1);
126
127 int px = x + (tile_x * SCALE);
128 int py = y + (tile_y * SCALE);
129
130 Color color = TILE_COLORS[color_id];
131 for (int sy = 0; sy < SCALE; sy++) {
132 for (int sx = 0; sx < SCALE; sx++) {
133 ImageDrawPixel(&tile_image, px + sx, py + sy, color);
134 }
135 }
136 }
137 }
138}

146 const uint8_t *tile_data = tile_buffer.data();
147
148 ImageClearBackground(&tile_image, (Color){17, 17, 17, 255});
149
150 for (int row = 0; row < TILES_PER_COL; row++) {
151 for (int col = 0; col < TILES_PER_ROW; col++) {
152 int tile_num = row * TILES_PER_ROW + col;

195void RayboyUI::drawStatsPanel() {
196 if (!show_stats)
197 return;
198
199 int panel_height = 300;
200 int panel_width = TILES_PER_ROW * TILE_SIZE * SCALE;
201 drawPanel(stats_panel_x, stats_panel_y, panel_width, panel_height,
202 "Performance Statistics");
203
204 int text_y = stats_panel_y + 40;
205 const int line_height = 18;
206
207 // 1. Fetch CPU Stats
208 if (shared->cpu_stats_ready.load(std::memory_order_acquire)) {
209 int r_idx = shared->cpu_stats_read_index.load(std::memory_order_acquire);
210 CPUStats current_stats = shared->cpu_stats[r_idx];
211
212 // Calculate MHz: (Cycles / Microseconds)
213 // Original GB: 4,194,304 Hz = ~4.19 MHz
214 double mhz = (double)current_stats.cycles / current_stats.time_us;
215 double percentage = (mhz / 4.194304) * 100.0;
216
217 // Calculate MIPS (Millions of Instructions Per Second)
218 double mips = (double)current_stats.instructions / current_stats.time_us;
219
220 // Draw CPU Performance
221 DrawText("--- CPU Engine ---", stats_panel_x + 10, text_y, 12,
222 ACCENT_COLOR);
223 text_y += line_height;
224
225 DrawText(TextFormat("Speed: %.3f MHz (%.1f%%)", mhz, percentage),
226 stats_panel_x + 15, text_y, 12, TEXT_COLOR);
227 text_y += line_height;
228
229 DrawText(TextFormat("Throughput: %.2f MIPS", mips), stats_panel_x + 15,
230 text_y, 12, TEXT_COLOR);
231 text_y += line_height;
232
233 DrawText(TextFormat("Cycles/Inst: %.2f", (double)current_stats.cycles /
234 current_stats.instructions),
235 stats_panel_x + 15, text_y, 12, TEXT_COLOR);
