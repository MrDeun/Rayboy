96static void proc_add(cpu_context *ctx) {
97 reg_type r = ctx->cur_instruction->reg1;
98
99 /* =========================
100 ADD SP, e8 (SPECIAL CASE)
101 ========================= */
102 if (r == RT_SP) {
103 uint16_t sp = ctx->regs.SP;
104 int8_t e = (int8_t)ctx->fetch_data;
105
106 uint16_t result = sp + e;
107
108 int h = ((sp & 0xF) + (e & 0xF)) > 0xF;
109 int c = ((sp & 0xFF) + (e & 0xFF)) > 0xFF;
110
111 ctx->regs.SP = result;
112
113 cpu_set_flags(ctx, 0, 0, h, c);
114 return;
115 }
116
117 bool is16bit = is_16bit(r);
118
119 /* =========================
120 ADD HL, rr
121 ========================= */
122 if (is16bit) {
123 emu_cycles(1);
124
125 uint32_t a = cpu_read_reg(r);
126 uint32_t b = ctx->fetch_data; // decoder must supply correct 16-bit value
127
128 uint32_t result = a + b;
129
130 int h = ((a & 0xFFF) + (b & 0xFFF)) > 0xFFF;
131 int c = result > 0xFFFF;
132
133 cpu_set_reg(r, result & 0xFFFF);
134 cpu_set_flags(ctx, -1, 0, h, c);
135 return;
136 }
137
138 /* =========================
139 ADD A, r / ADD A, n
140 ========================= */
141 uint8_t a = ctx->regs.A;
142 uint8_t b = ctx->fetch_data & 0xFF;
143
144 uint16_t result = a + b;
145
146 int z = (result & 0xFF) == 0;
147 int h = ((a & 0xF) + (b & 0xF)) > 0xF;
148 int c = result > 0xFF;
149
150 ctx->regs.A = result & 0xFF;
151 cpu_set_flags(ctx, z, 0, h, c);
152}

247static void proc_cb(cpu_context *ctx) {
248 uint8_t op = ctx->fetch_data;
249 reg_type reg = decode_reg(op & 0b111);
250 uint8_t bit = (op >> 3) & 0b111;
251 uint8_t bit_op = (op >> 6) & 0b11;
252 uint8_t reg_val = cpu_read_reg8(reg);
253
254 emu_cycles(1);
255
256 if (reg == RT_HL) {
257 emu_cycles(2);
258 }
259
260 switch (bit_op) {
261 case 1:
262 // BIT
263 cpu_set_flags(ctx, !(reg_val & (1 << bit)), 0, 1, -1);
264 return;
265
266 case 2:
267 // RST
268 reg_val &= ~(1 << bit);
269 cpu_set_reg8(reg, reg_val);
270 return;
271
272 case 3:
273 // SET
274 reg_val |= (1 << bit);
275 cpu_set_reg8(reg, reg_val);
276 return;
277 }
278
279 bool flagC = CPU_FLAG_C;
280
281 switch (bit) {
282 case 0: {
283 // RLC
284 bool setC = false;
285 uint8_t result = (reg_val << 1) & 0xFF;
286
287 if ((reg_val & (1 << 7)) != 0) {
288 result |= 1;
289 setC = true;
290 }
291
292 cpu_set_reg8(reg, result);
293 cpu_set_flags(ctx, result == 0, false, false, setC);
294 }
295 return;
296
297 case 1: {
298 // RRC
299 uint8_t old = reg_val;
300 reg_val >>= 1;
301 reg_val |= (old << 7);
302
303 cpu_set_reg8(reg, reg_val);
304 cpu_set_flags(ctx, !reg_val, false, false, old & 1);
305 }
306 return;
307
308 case 2: {
309 // RL
310 uint8_t old = reg_val;
311 reg_val <<= 1;
312 reg_val |= flagC;
313
314 cpu_set_reg8(reg, reg_val);
315 cpu_set_flags(ctx, !reg_val, false, false, !!(old & 0x80));
316 }
317 return;
318
319 case 3: {
320 // RR
321 uint8_t old = reg_val;
322 reg_val >>= 1;
323
324 reg_val |= (flagC << 7);
325
326 cpu_set_reg8(reg, reg_val);
327 cpu_set_flags(ctx, !reg_val, false, false, old & 1);
328 }
329 return;
330
331 case 4: {
332 // SLA
333 uint8_t old = reg_val;
334 reg_val <<= 1;
335
336 cpu_set_reg8(reg, reg_val);
337 cpu_set_flags(ctx, !reg_val, false, false, !!(old & 0x80));
338 }
339 return;
340
341 case 5: {
342 // SRA
343 uint8_t u = (int8_t)reg_val >> 1;
344 cpu_set_reg8(reg, u);
345 cpu_set_flags(ctx, !u, 0, 0, reg_val & 1);
346 }
347 return;
348
349 case 6: {
350 // SWAP
351 reg_val = ((reg_val & 0xF0) >> 4) | ((reg_val & 0xF) << 4);
352 cpu_set_reg8(reg, reg_val);
353 cpu_set_flags(ctx, reg_val == 0, false, false, false);
354 }
355 return;
356
357 case 7: {
358 // SRL
359 uint8_t u = reg_val >> 1;
360 cpu_set_reg8(reg, u);
361 cpu_set_flags(ctx, !u, 0, 0, reg_val & 1);
362 }
363 return;
364 }
365 auto err = fmt::format("ERROR: INVALID CB: {:X}", op);
366 ERROR(err.c_str());
367}
